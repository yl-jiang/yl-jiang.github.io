class的定义

#include<iostream>
#include<vector>
#include<string>
#include<algorithm>

using namespace std;

class Stack{
    public:
        // 一下这些函数只是声明
        bool push(const string &);
        bool pop(string &elem);
        bool peek(string &elem);
        bool empty();
        bool full();
        bool findit(string &elem);
        int countit(string &elem);

        // size函数定义在class之内
        int size(){
            return _stack.size();
        }

    private:
        vector<string> _stack;
};

inline bool Stack::empty(){
    return _stack.empty();
}

inline bool Stack::pop(string &elem){
    if (empty())  // empty()必须再调用之前已经被定义
        return false;
    elem = _stack.back();
    _stack.pop_back();
    return true;
}

inline bool Stack::full(){
    return _stack.size() == _stack.max_size();
}

inline bool Stack::peek(string &elem){
    if(empty())
        return false;
    elem = _stack.back();
    return true;
}

inline bool Stack::push(const string &elem){
    if (full())
        return false;
    _stack.push_back(elem);
    return true;
}

inline bool Stack::findit(string &elem){
    if(empty())
        return false;
    if (find(_stack.begin(), _stack.end(), elem) != _stack.end())
        return true;
}

inline int Stack::countit(string &elem){
    if (empty())
        return 0;
    return count(_stack.begin(), _stack.end(), elem);
}
  • member function都必须在class主体内进行声明
  • 可以直接在class主体内定义member function，就像函数size()一样
  • 也可以在class主体外定义函数，只是需要注意：
    • 当把class主体和member function都放在”.h”文件内时，member可以使用inline的形式定义，即：inline classname::functionname
    • 当要使用non-inline的形式定义member function，那么该member function建议定义在程序文件，即：.cpp中
  • 在class主体之外声明的member可以选择使用[或者不使用]inline function形式，例如：empty()函数使用inline方式，而full()函数则没有使用

class object的使用

我们一般将class的定义放在一个'.h'的文件中，例如我们把上面这段程序命名为'stack.h'，并在同一目录下新建一个'stack.cpp'的包含main()函数的程序文件：

#include "stack.h"

int main(){
    Stack stack;
    string str[] = {"hello", "word", "how", "are", "you"};
    string tmp;
    string find_str = "hello";

    for (unsigned int ix = 0; ix < 5; ++ix){
        if (stack.push(str[ix])){
            cout << "push word : " << str[ix] << " in stack" << endl;
        }
    }

    if (stack.findit(find_str))
        cout << " find it!" << endl;

    cout << "word # " << find_str << " occur : " << stack.countit(find_str) << " times" << endl;

    while(stack.size()){
    if (stack.pop(tmp)){
        cout << tmp << " ";
    }
    }

    getchar();
}

被定义为const的member function在class object被初始化后不会更改class object中private内定义的各data member。

如果上面的class有一个member function定义如下的话，是无法通过编译的：

int next_plus_one() const {
            return ++_next; // 错误，const member function是不能修改class object的
        }

Mutable Data Member

定义为可变数据成员的变量(关键字为: mutable)，对其进行的修改不会破坏class object的常量性。

#include<iostream>
#include<string>
#include<vector>

using namespace std;

class Triangular{

    public:
        // class initilization
        Triangular(): _length(0), _beg_pos(0), _next(0){}
        Triangular(int len, int pos);

        //const member function
        int get_len() const {
            return _length;
        }
        int get_cur_pos() const;
        int get_next_pos() const;
        int next_plus_one() const {
            return ++_next;  // 虽然该函数为const function member，但是它可以修改private中定义的_next
        }

    private:
        int _length;
        int _beg_pos;
        mutable int _next;  // 此处定义_next为可变数据成员
};

// class initilization funtion

    _length = len > 0 ? len : 0;
    _beg_pos = pos > 0 ? pos : 0;
    _next = pos - 1;
}

    return _beg_pos;
}

int Triangular::get_next_pos() const {
    return _next;
}

    cout << "_length = " << obj.get_len()
         << "\n_beg_pos = " << obj.get_cur_pos()
         << "\n_next = " << obj.get_next_pos() << endl;

static data member

static成员变量属于类，不属于某个具体的对象，即使创建多个对象，也只为static成员变量分配一份内存，所有对象使用的都是这份内存中的数据。当某个对象修改了该static成员变量，也会影响到其他对象。

静态成员属于全局变量，是所有实例化以后的对象所共享的，显然这种共有对象，不能在任何函数和局部作用域中初始化。

static data member的初始化和简单使用：

#include<iostream>
#include<string>
#include<vector>

using namespace std;

class Triangular{

    public:
        // class initialization
        Triangular();

        //const member function
        int get_len() const {
            return _length;
        }
        int get_cur_pos() const;
        int get_next_pos() const;

        void set_static_member(int val) {
            static_int = val;
        }
        int get_static_val() {
            return static_int;
        }

    private:
        int _length;
        int _beg_pos;
        int _next;
        static int static_int;
};

// constructor functions
Triangular::Triangular(int len, int pos){
    _length = len > 0 ? len : 0;
    _beg_pos = pos > 0 ? pos : 0;
    _next = pos - 1;
}
Triangular::Triangular(){
    _length = 0;
    _beg_pos = 0;
    _next = 0;
}

// define const member function out of class
int Triangular::get_cur_pos() const {
    return _beg_pos;
}

int Triangular::get_next_pos() const {
    return _next;
}

// initialize static data member 
int Triangular::static_int = 10;

int main(){
    Triangular obj(10, 1);
    obj.set_static_member(3);
    cout << "obj_static_value = " << obj.get_static_val() << endl;

    Triangular obj_2;

    getchar();
}

static member function

那些不访问任何non-static member的member function才可以被声明为static，声明方式为在之前加上关键字static。当我们在class主体外部对static member function进行定义时，无需重复加上关键字static(该规则同样适用于static data member)。

#include<iostream>
#include<string>
#include<vector>
#include<algorithm>
#include<iterator>
using namespace std;


class Triangular{

    public:
        static bool is_elem(int);
        static void gen_elems(int length);
        static void gen_elems_to_value(int val);
        static void display(int length, int beg_pos, ostream &os = cout);

    private:
        static const int _max_elems = 1024;
        static vector<int> _elems;
};

// 在class外部对static data member进行定义时可以不加static关键字
vector<int> Triangular::_elems;

// the member function declared as static in class, you do not need to add the 'static' keyword when you define the member function out of class
void Triangular::gen_elems_to_value(int val){
    int ix = _elems.size();
    if (!ix){
        _elems.push_back(1);
        ix = 1;
    }

    while (_elems[ix - 1] < val && ix < _max_elems){
        cout << "elems to value: " << ix * (ix + 1) / 2 << endl;
        ++ix;
        _elems.push_back(ix * (ix + 1) / 2);
    }

    if (ix == _max_elems){
        cerr << "Triangular Sequence: oops: value too large"
             << val << " -- exceeds max size of "
             << _max_elems << endl;
    }
}


bool Triangular::is_elem(int elem){
    if (!_elems.size() || _elems[_elems.size() - 1] < elem){
        gen_elems_to_value(elem);
    }

    vector<int>::iterator found_it;
    vector<int>::iterator end_it = _elems.end();
    found_it = find(_elems.begin(), end_it, elem);
    return found_it != end_it;
}

void Triangular::display(int length, int beg_pos, ostream &os){
    if (!_elems.size())
        _elems.push_back(1);
    int ix = _elems.size();
    while (ix < length){
        ++ix;
        _elems.push_back(ix * (ix + 1) / 2);
    }
    while (ix > length){
        _elems.pop_back();
        --ix;
    }

    if (beg_pos > length)
        cerr << "param 'beg_pos' should less than param 'length'" << endl;

    for (int i = beg_pos - 1; i <= length - 1; ++i)
        os << _elems[i] << " ";
    os << endl;
}


int main(){
    int val = 15;
    if (Triangular::is_elem(val))  // 调用static member function
        cout << "the value: " << val << " in the Triangular" << endl;
    else
        cout << "the value: " << val << " not in the Triangular" << endl;
    Triangular::display(10, 1);   // 调用static member function

    getchar();
}

一般地，在调用non-static member function时那样，必须先定义一个该class的实例，然后再通过实例去调用类里面的函数。而调用static member function，只需在static member function前加上class scope就可以直接调用该函数。

有关class的初始化注意事项

初始化位置

一般都是类内声明，类外初始化，因为静态成员属于整个类，而不属于某个对象，如果在类内初始化，会导致每个对象都包含该静态成员，这是矛盾的。不过，静态常量成员是可以在类内声明的：

class Student {
    public:
        // ...
    private:
        static int age = 20;//错误，静态成员不能在类内初始化
        const int age = 20;//错误，常量成员也不能在类内初始化
        static const int age = 20;//正确
}

初始化列表初始顺序

#include <iostream>
using namespace std;

class Base 
{
public:
    Base(int i) : m_j(i), m_i(m_j) {}
    Base() : m_j(0), m_i(m_j) {}
    int get_i() const
    {
        return m_i;
    }
    int get_j() const
    {
        return m_j;
    }

private:
    int m_i;
    int m_j;

};

int main()
{
    Base obj(98);
    cout << obj.get_i() << endl << obj.get_j() << endl;
    return 0;
}

输出为:一个随机数和数字98。

为什么呢？因为对于初始化列表而言，对成员变量的初始化，是严格按照声明次序(private部分)，而不是在初始化列表中的顺序进行初始化，如果改为赋值初始化则不会出现这个问题，当然，为了使用初始化列表，还是严格注意声明顺序吧，比如先声明数组大小，再声明数组这样。

    cout << "Triangular Series of " << tri.length() << " elements: " << endl;
    cout << tri << endl;
    while (it != end_it){
        cout << *it << " ";
        ++it;
    }
    cout << endl;

    getchar();
}

自定义function object

function object为“提供了function call(即‘operator()’ memberfunction)”运算符的class。【相当于Python中实现了__call__()方法的class】

栗子：定义一个提供了function call运算符的class

#include<iostream>
#include<vector>
#include<string>
#include<algorithm>

using namespace std;

class LessThan
{
    public:
        LessThan(int val): _val(val){}
        int comp_val() const { return _val; }
        void comp_val(int nval) { _val = nval; }

        bool operator()(int _value) const;  // 相当于实现Python中的__call__()方法

    private:
        int _val;
};

上面的代码就自定义了一个function object：LessThan

使用自定义function object的一般形式：

//  一般情况下使用自定义的function object
bool LessThan::operator()(int val) const{
    return val < _val;
}

int count_less_than(const vector<int> &vec, int comp)
{
    LessThan lt(comp);  // 声明一个自定义的function object

    int count = 0;
    for (vector<int>::const_iterator iter = vec.begin(); iter != vec.end(); ++iter){
        if (lt(*iter)) // 使用自定义的function object
            ++count;
    }
    return count;
}

更常用的情况是，我们将function object作为一个参数传给泛型算法：

//  把自定义的function object当作function的参数
void print_less_than(const vector<int> &vec, int comp_val, ostream &os=cout){
    LessThan lt(comp_val);

    vector<int>::const_iterator iter = vec.begin();
    vector<int>::const_iterator iter_end = vec.end();

    os << "elements less than " << lt.comp_val() << endl;
    while ((iter = find_if(iter, iter_end, lt)) != iter_end){
        os << *iter << " ";
        ++iter;
    }
}

验证上面的程序：

int main(){
    int ia[] = {17, 12, 44, 9, 18, 45, 6, 14, 23, 67, 9, 0, 27, 55, 8, 16};
    vector<int> vec(ia, ia + 16);
    int comp_val = 20;

    cout << "Number of elements less than " << comp_val << " are" << count_less_than(vec, comp_val) << endl;
    print_less_than(vec, comp_val);
    getchar();
}

重载iostream运算符

重载该运算符，以实现对某个class object的读取和写入操作。【相当于Python中实现了__repr__()方法的class】

ostream& operator<<(ostream &os, Triangular &rhs){
    os << "Triangular(" << rhs.length() << " ," << rhs.beg_pos() << ")\n";
    rhs.display(rhs.length(), rhs.beg_pos());
    return os;
}

将传入的ostream又原封不动的返回，这样可以串接多个output运算符。传入的ostream并没有声明为const，因为每个output都会对ostream对象的内部状态进行修改。

为什么不把output运算设计为一个member function？

因为作为一个member function，其左操作数必须是隶属于同一个class的对象。如果在上面的栗子中，output被设计为一个member function：

Triangular tri(8, 3);

// 输出tri
tri << cout << "\n" ;  // 其中'tri <<'会调用‘operator <<()'函数

指向member function的指针

class num_sequence{
    public:
        typedef void (num_sequence::*PtrType)(int);

        void fibonacci(int);
        void pell(int);
        void lucas(int);
        void triangular(int);
        void sequare(int);
        // ...

    private:
        PtrType _pmf;
};

PtrType声明为一个指针，指向num_sequence的member function，后者的返回类型是void，只接受一个int参数。

member function的调用

num_sequence ns;
num_sequence *pns = &ns;

假设num_sequence拥有member function pm，调用该function可以有以下两种方法：

  • ns.pm(pos);
  • pns->pm(pos);

直接定义一个类对象ns，调用member function使用：'.'。定义一个指向类对象的指针，调用member function使用：'->'。
